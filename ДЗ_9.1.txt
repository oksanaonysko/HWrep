function findValueByKey(companyName) {
  function search(company, name) {
    if (company.name === name) {
      console.log(`Назва: ${company.name}`);
      console.log(`Тип компанії: ${company.type}`);
      console.log(`Використовує: ${company.uses}`);
      console.log(`Пропонує: ${company.sells}`);
      return true; 
    }
    
    if (company.clients) {
      for (let client of company.clients) {
        if (search(client, name)) {
          return true; 
        }
      }
    }
    
    if (company.partners) {
      for (let partner of company.partners) {
        if (search(partner, name)) {
          return true;  
        }
      }
    }
    return false; 
  }
  
  if (!search(company, companyName)) {
    console.log(`Будь-ласка, перевірте назву компанії.`);
  }
}
